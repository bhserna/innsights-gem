Overview

Stores the record instance that holds the key values that will be sent to the service the way to get these values from within the record is configured using the watch DSL

def initialize(record, report)
  @record = record
  @report = report
  @name = attr_call(report.name)
  @user = attr_call(report.report_user)
  @group = attr_call(report.report_group)
  @created_at = attr_call(report.created_at_method)
  @metrics = fetch_metrics(report.metrics)
end
